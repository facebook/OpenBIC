217static inline int calculate_accurate_MBR(uint8_t sensor_num, int val)
218{ // for better accuracy, enlarge SDR to two byte scale
219 if (SDR_M(sensor_num) == 0) {
220 return ((val << 8) * SDR_Rexp(sensor_num));
221 }
222 return ((val << 8) / SDR_M(sensor_num) * SDR_Rexp(sensor_num));
223}
224
225static inline int calculate_MBR(uint8_t sensor_num, int val)
226{
227 if (SDR_M(sensor_num) == 0) {
228 return (val * SDR_Rexp(sensor_num) + round_add(sensor_num, val));
229 }
230 return (val * SDR_Rexp(sensor_num) / SDR_M(sensor_num) + round_add(sensor_num, val));
231}
232
233static inline float convert_MBR_to_reading(uint8_t sensor_num, uint8_t val)
234{
235 if (SDR_M(sensor_num) == 0) {
236 return (val - round_add(sensor_num, val)) / SDR_Rexp(sensor_num);
237 }
238
239 return (val - round_add(sensor_num, val)) * SDR_M(sensor_num) / SDR_Rexp(sensor_num);
240}
